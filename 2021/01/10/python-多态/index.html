多态性指的是可以在不考虑对象具体类型的情况下而直接使用对象

class Animal: # 统一所有子类的方法
def say(self):
print(‘动物基本的发声频率。。。’,end=’ ‘)

class People(Animal):
def say(self):
super().say()
print(‘嘤嘤嘤嘤嘤嘤嘤’)

class Dog(Animal):
def say(self):
super().say()
print(‘汪汪汪’)

class Pig(Animal):
def say(self):
super().say()
print(‘哼哼哼’)
